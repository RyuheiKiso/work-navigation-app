# 06 状態管理実装設計

本章は 3 つの APP（ハンディ APP / マスタメンテ APP / 管理コンソール）における状態管理アーキテクチャを確定する。各 APP の特性（オフライン対応要否・サーバー状態同期頻度・undo/redo 要否）に応じた実装方針と TypeScript 型定義を示す。

---

## 1. 状態管理方針の選定根拠

| APP | 採用技術 | 選定根拠 |
|---|---|---|
| ハンディ APP | React Context + useReducer | オフライン時も動作、外部依存最小、React Native 標準機能で完結 |
| マスタメンテ APP | Zustand（UI 状態）+ react-query（サーバー状態）| SOP 編集の undo/redo スタック管理に Zustand が適合、サーバー同期は react-query が担当 |
| 管理コンソール | react-query（サーバー状態）+ Zustand（UI 最小限）| ほぼ全状態がサーバー由来のダッシュボード系、Zustand は UI 設定のみ保持 |

### 2-1. AuthContext

```typescript
// apps/handy/src/contexts/AuthContext.tsx

export interface AuthUser {
  userId: string;
  displayName: string;
  role: 'operator' | 'supervisor' | 'quality_admin';
  locale: 'ja' | 'en' | 'zh';
}

export interface AuthState {
  token: string | null;
  user: AuthUser | null;
  isAuthenticated: boolean;
  isLoading: boolean;
  error: string | null;
}

export type AuthAction =
  | { type: 'LOGIN_START' }
  | { type: 'LOGIN_SUCCESS'; payload: { token: string; user: AuthUser } }
  | { type: 'LOGIN_FAILURE'; payload: { errCode: string; message: string } }
  | { type: 'LOGOUT' };

export const authReducer = (state: AuthState, action: AuthAction): AuthState => {
  switch (action.type) {
    case 'LOGIN_START':
      return { ...state, isLoading: true, error: null };
    case 'LOGIN_SUCCESS':
      return {
        ...state,
        isLoading: false,
        token: action.payload.token,
        user: action.payload.user,
        isAuthenticated: true,
        error: null,
      };
    case 'LOGIN_FAILURE':
      return {
        ...state,
        isLoading: false,
        error: action.payload.message,
        isAuthenticated: false,
      };
    case 'LOGOUT':
      return {
        token: null,
        user: null,
        isAuthenticated: false,
        isLoading: false,
        error: null,
      };
    default:
      return state;
  }
};
```

### 2-2. NetworkContext

```typescript
// apps/handy/src/contexts/NetworkContext.tsx

export interface NetworkState {
  isOnline: boolean;
  isEmergencyMode: boolean;       // オフライン時の縮退動作モード
  lastOnlineAt: Date | null;
  pendingSyncCount: number;        // SQLite ローカルキュー未同期件数
}

export type NetworkAction =
  | { type: 'SET_ONLINE' }
  | { type: 'SET_OFFLINE' }
  | { type: 'INCREMENT_PENDING' }
  | { type: 'SYNC_COMPLETE'; payload: { syncedCount: number } };

export const networkReducer = (
  state: NetworkState,
  action: NetworkAction,
): NetworkState => {
  switch (action.type) {
    case 'SET_ONLINE':
      return {
        ...state,
        isOnline: true,
        isEmergencyMode: false,
        lastOnlineAt: new Date(),
      };
    case 'SET_OFFLINE':
      return { ...state, isOnline: false, isEmergencyMode: true };
    case 'INCREMENT_PENDING':
      return { ...state, pendingSyncCount: state.pendingSyncCount + 1 };
    case 'SYNC_COMPLETE':
      return {
        ...state,
        pendingSyncCount: Math.max(
          0,
          state.pendingSyncCount - action.payload.syncedCount,
        ),
      };
    default:
      return state;
  }
};
```

### 2-3. StepExecutionContext

```typescript
// apps/handy/src/contexts/StepExecutionContext.tsx

export type StepStatus =
  | 'PENDING'
  | 'IN_PROGRESS'
  | 'COMPLETED'
  | 'SKIPPED'
  | 'FAILED';

export type ExecutionStatus =
  | 'NOT_STARTED'
  | 'IN_PROGRESS'
  | 'SUSPENDED'
  | 'COMPLETED';

export interface StepExecutionState {
  executionId: string | null;
  sopId: string | null;
  currentStepIndex: number;
  totalSteps: number;
  status: ExecutionStatus;
  stepStatuses: Record<string, StepStatus>;
  evidenceIds: string[];
  measurementIds: string[];
  signatureId: string | null;
  suspendedAt: Date | null;
  suspendReason: string | null;
}

export type StepExecutionAction =
  | { type: 'INIT_EXECUTION'; payload: { executionId: string; sopId: string; totalSteps: number } }
  | { type: 'COMPLETE_STEP'; payload: { stepId: string } }
  | { type: 'ADVANCE_STEP' }
  | { type: 'SUSPEND'; payload: { reason: string; comment?: string } }
  | { type: 'RESUME'; payload: { executionId: string } }
  | { type: 'FINALIZE' }
  | { type: 'ADD_EVIDENCE'; payload: { evidenceId: string } }
  | { type: 'ADD_MEASUREMENT'; payload: { measurementId: string } }
  | { type: 'SET_SIGNATURE'; payload: { signatureId: string } };

export const stepExecutionReducer = (
  state: StepExecutionState,
  action: StepExecutionAction,
): StepExecutionState => {
  switch (action.type) {
    case 'INIT_EXECUTION':
      return {
        ...state,
        executionId: action.payload.executionId,
        sopId: action.payload.sopId,
        totalSteps: action.payload.totalSteps,
        currentStepIndex: 0,
        status: 'IN_PROGRESS',
      };
    case 'COMPLETE_STEP':
      return {
        ...state,
        stepStatuses: {
          ...state.stepStatuses,
          [action.payload.stepId]: 'COMPLETED',
        },
      };
    case 'ADVANCE_STEP':
      return {
        ...state,
        currentStepIndex: state.currentStepIndex + 1,
      };
    case 'SUSPEND':
      return {
        ...state,
        status: 'SUSPENDED',
        suspendedAt: new Date(),
        suspendReason: action.payload.reason,
      };
    case 'RESUME':
      return {
        ...state,
        status: 'IN_PROGRESS',
        suspendedAt: null,
        suspendReason: null,
      };
    case 'FINALIZE':
      return { ...state, status: 'COMPLETED' };
    case 'ADD_EVIDENCE':
      return {
        ...state,
        evidenceIds: [...state.evidenceIds, action.payload.evidenceId],
      };
    case 'ADD_MEASUREMENT':
      return {
        ...state,
        measurementIds: [...state.measurementIds, action.payload.measurementId],
      };
    case 'SET_SIGNATURE':
      return { ...state, signatureId: action.payload.signatureId };
    default:
      return state;
  }
};
```

## 3. マスタメンテ APP 状態管理

### 3-1. SopEditorStore（Zustand）

```typescript
// apps/master/src/stores/sopEditorStore.ts

export interface StepDraft {
  id: string;
  stepNumber: number;
  type: 'STANDARD' | 'BRANCH' | 'CUSTOM';
  instruction: Record<'ja' | 'en' | 'zh', string>;
  dslCondition: unknown | null;           // JSON Logic DSL オブジェクト（BRANCH のみ）
  mediaRequired: boolean;
  measurementRequired: boolean;
  signatureRequired: boolean;
}

export interface SopEditorStore {
  // --- 状態 ---
  sopId: string | null;
  steps: StepDraft[];
  isDirty: boolean;
  lastSavedAt: Date | null;
  undoStack: StepDraft[][];
  redoStack: StepDraft[][];
  activeStepId: string | null;

  // --- アクション ---
  initEditor: (sopId: string, steps: StepDraft[]) => void;
  addStep: (afterIndex?: number) => void;
  updateStep: (id: string, patch: Partial<StepDraft>) => void;
  removeStep: (id: string) => void;
  reorderStep: (fromIndex: number, toIndex: number) => void;
  undo: () => void;
  redo: () => void;
  markClean: (savedAt: Date) => void;
  setActiveStep: (id: string | null) => void;
}

export const useSopEditorStore = create<SopEditorStore>((set, get) => ({
  sopId: null,
  steps: [],
  isDirty: false,
  lastSavedAt: null,
  undoStack: [],
  redoStack: [],
  activeStepId: null,

  initEditor: (sopId, steps) =>
    set({ sopId, steps, isDirty: false, undoStack: [], redoStack: [] }),

  addStep: (afterIndex) =>
    set((state) => {
      const newStep: StepDraft = {
        id: crypto.randomUUID(),
        stepNumber: state.steps.length + 1,
        type: 'STANDARD',
        instruction: { ja: '', en: '', zh: '' },
        dslCondition: null,
        mediaRequired: false,
        measurementRequired: false,
        signatureRequired: false,
      };
      const insertAt =
        afterIndex !== undefined ? afterIndex + 1 : state.steps.length;
      const newSteps = [
        ...state.steps.slice(0, insertAt),
        newStep,
        ...state.steps.slice(insertAt),
      ];
      return {
        undoStack: [...state.undoStack, state.steps],
        redoStack: [],
        steps: newSteps,
        isDirty: true,
      };
    }),

  updateStep: (id, patch) =>
    set((state) => ({
      undoStack: [...state.undoStack, state.steps],
      redoStack: [],
      steps: state.steps.map((s) => (s.id === id ? { ...s, ...patch } : s)),
      isDirty: true,
    })),

  removeStep: (id) =>
    set((state) => ({
      undoStack: [...state.undoStack, state.steps],
      redoStack: [],
      steps: state.steps.filter((s) => s.id !== id),
      isDirty: true,
    })),

  reorderStep: (fromIndex, toIndex) =>
    set((state) => {
      const newSteps = [...state.steps];
      const [moved] = newSteps.splice(fromIndex, 1);
      newSteps.splice(toIndex, 0, moved);
      return {
        undoStack: [...state.undoStack, state.steps],
        redoStack: [],
        steps: newSteps,
        isDirty: true,
      };
    }),

  undo: () =>
    set((state) => {
      if (state.undoStack.length === 0) return state;
      const prevSteps = state.undoStack[state.undoStack.length - 1];
      return {
        steps: prevSteps,
        undoStack: state.undoStack.slice(0, -1),
        redoStack: [...state.redoStack, state.steps],
        isDirty: true,
      };
    }),

  redo: () =>
    set((state) => {
      if (state.redoStack.length === 0) return state;
      const nextSteps = state.redoStack[state.redoStack.length - 1];
      return {
        steps: nextSteps,
        redoStack: state.redoStack.slice(0, -1),
        undoStack: [...state.undoStack, state.steps],
        isDirty: true,
      };
    }),

  markClean: (savedAt) =>
    set({ isDirty: false, lastSavedAt: savedAt }),

  setActiveStep: (id) =>
    set({ activeStepId: id }),
}));
```

## 5. ダークモード状態管理（全 APP 共通）

```typescript
// packages/theme/src/useThemeMode.ts

/**
 * 夜勤帯（20:00-翌 6:00）はダークモード自動適用（FR-UI-004）。
 * CFG から時間帯設定を取得する。
 */
export const useThemeMode = (): 'light' | 'dark' => {
  const [mode, setMode] = React.useState<'light' | 'dark'>(() =>
    detectMode(new Date().getHours()),
  );

  React.useEffect(() => {
    const id = setInterval(() => {
      setMode(detectMode(new Date().getHours()));
    }, 60_000); // 1 分ごとに評価
    return () => clearInterval(id);
  }, []);

  return mode;
};

const detectMode = (hour: number): 'light' | 'dark' => {
  const start = Number(import.meta.env.VITE_DARK_MODE_START ?? '20');
  const end   = Number(import.meta.env.VITE_DARK_MODE_END   ?? '6');
  return hour >= start || hour < end ? 'dark' : 'light';
};
```

## 6. SQLite ローカルストレージ（ハンディ APP オフライン対応）

ハンディ APP は TypeORM + SQLite をローカルストレージとして使用し、オフライン時のイベント記録とオンライン復帰後の Outbox 同期を担う。

```typescript
// apps/handy/src/db/LocalEventQueue.ts

export interface LocalWorkEvent {
  localId: number;               // SQLite AUTOINCREMENT
  eventId: string;               // UUID（サーバー送信用）
  eventType: string;             // StepCompleted / MediaCapture 等
  payload: string;               // JSON シリアライズ済み
  occurredAt: string;            // ISO 8601
  syncStatus: 'PENDING' | 'SYNCED' | 'FAILED';
  retryCount: number;
}

/**
 * ネットワーク復帰時に PENDING イベントをサーバーへ同期する。
 * 成功した場合は syncStatus を SYNCED に更新する。
 * FAILED が 3 回を超えた場合は DLQ 扱いとしてローカルに保持する。
 */
export const syncPendingEvents = async (
  repository: Repository<LocalWorkEvent>,
  apiClient: ApiClient,
): Promise<void> => {
  const pending = await repository.find({
    where: { syncStatus: 'PENDING' },
    order: { occurredAt: 'ASC' },
  });

  for (const event of pending) {
    try {
      await apiClient.postWorkEvent(JSON.parse(event.payload));
      await repository.update(event.localId, { syncStatus: 'SYNCED' });
    } catch {
      const nextRetry = event.retryCount + 1;
      await repository.update(event.localId, {
        retryCount: nextRetry,
        syncStatus: nextRetry >= 3 ? 'FAILED' : 'PENDING',
      });
    }
  }
};
```

---

**本節で確定した方針**
- **ハンディ APP は React Context + useReducer（AuthContext / NetworkContext / StepExecutionContext）を採用し、外部ライブラリ非依存・オフライン完全動作を確定した。**
- **マスタメンテ APP の SOP 編集は Zustand の SopEditorStore（undo/redo スタック管理）と react-query の自動保存 mutation を組み合わせ、データ損失ゼロと操作取消を同時に担保することを確定した。**
- **管理コンソールは react-query をサーバー状態の主管として採用し、ポーリング間隔（refetchInterval）・staleTime・invalidation を型安全に制御する設計を確定した。**
